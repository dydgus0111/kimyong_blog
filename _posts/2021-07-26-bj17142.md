# 1. 문제
[17142 연구소3](https://www.acmicpc.net/problem/17142)
# 2. 코드
```{.java}
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.LinkedList;
import java.util.List;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {
	static BufferedReader br=new BufferedReader(new InputStreamReader(System.in));
	static int N;
	static int M;
	static int map[][];
	static class virus{
		int x;
		int y;
		public virus(int x, int y) {
			this.x=x;
			this.y=y;
		}
	}
	static class pair{
		int x;
		int y;
		int time;
		public pair(int x, int y,int time) {
			this.x=x;
			this.y=y;
			this.time=time;
		}
	}
	static int dx[]= {-1,1,0,0};
	static int dy[]= {0,0,-1,1};
	static List<virus> vlist;
	static int safecnt;
	public static void main(String[] args) throws NumberFormatException, IOException {
		StringTokenizer st=new StringTokenizer(br.readLine());
		N=Integer.parseInt(st.nextToken());
		M=Integer.parseInt(st.nextToken());
		vlist=new ArrayList<>();
		map=new int[N][N];
		ans=Integer.MAX_VALUE;
		safecnt=0;
		for(int i=0;i<N;i++) {
			st=new StringTokenizer(br.readLine());
			for(int j=0;j<N;j++){
				int input=Integer.parseInt(st.nextToken());
				map[i][j]=input;
				if(input==2) {
					vlist.add(new virus(i, j));
				}
				else if(input==0) {
					safecnt++;
				}
			}
		}
		combi(new int[M], 0, 0);
		if(ans==Integer.MAX_VALUE)
			System.out.println(-1);
		else
			System.out.println(ans);
	}
	static int ans;
	static void combi(int sel[], int n, int r) {
		if(r==sel.length) {
			bfs(sel, new boolean[N][N]);
			
			return;
		}
		if(n==vlist.size())
			return;
		sel[r]=n;
		combi(sel, n+1, r+1);
		combi(sel, n+1, r);
	}
	static boolean isRange(int x, int y) {
		return x>=0&&x<N&&y>=0&&y<N;
	}
	static void bfs(int sel[],boolean check[][]) {
		int scnt=0;
		int time=0;
		Queue<pair> queue=new LinkedList<>();
		for(int i=0;i<sel.length;i++) {
			int x=vlist.get(sel[i]).x;
			int y=vlist.get(sel[i]).y;
			queue.add(new pair(x, y,0));
			check[x][y]=true;
		}
		while(!queue.isEmpty()) {
			pair next=queue.poll();
			for(int i=0;i<4;i++) {
				int nx=next.x+dx[i];
				int ny=next.y+dy[i];
				if(isRange(nx, ny)&&map[nx][ny]!=1&&!check[nx][ny]) {
					if(map[nx][ny]==0) {
						scnt++;
						time=next.time+1;
					}
					check[nx][ny]=true;
					queue.add(new pair(nx, ny, next.time+1));
				}
			}
		}
		if(scnt==safecnt) {
			if(ans>time)
				ans=time;
		}
	}
}
```
