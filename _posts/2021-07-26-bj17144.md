# 1. 문제
[17144 미세먼지 안녕!](https://www.acmicpc.net/problem/17144)
# 2. 코드
```{.java}
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.LinkedList;
import java.util.List;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {
	static int map[][];
	static int R;
	static int C;
	static int T;
	static int dx[]= {-1,1,0,0};
	static int dy[]= {0,0,-1,1};
	static class dust{
		int x;
		int y;
		int quantity;
		public dust(int x, int y, int q) {
			this.x=x;
			this.y=y;
			this.quantity=q;
		}
		
	}
	static BufferedReader br=new BufferedReader(new InputStreamReader(System.in));
	static int upAirx;
	static int upAiry;
	static int downAirx;
	static int downAiry;
	static List<dust> dlist;
	public static void main(String[] args) throws IOException {
		StringTokenizer st=new StringTokenizer(br.readLine());
		R=Integer.parseInt(st.nextToken());
		C=Integer.parseInt(st.nextToken());
		T=Integer.parseInt(st.nextToken());
		map=new int[R][C];
		upAirx=-1;
		upAiry=-1;
		downAirx=-1;
		downAiry=-1;
		dlist=new ArrayList<dust>();
		for(int i=0;i<R;i++) {
			st=new StringTokenizer(br.readLine());
			for(int j=0;j<C;j++) {
				int input=Integer.parseInt(st.nextToken());
				map[i][j]=input;
				if(input==-1&&upAirx==-1&&upAiry==-1) {
					upAirx=i;
					upAiry=j;
				}
				else if(input==-1&&upAirx!=-1&&upAiry!=-1) {
					downAirx=i;
					downAiry=j;
				}
			}
		}
		while(T!=0) {
			int cmap[][]=new int[R][C];
			for(int i=0;i<R;i++) {
				cmap[i]=Arrays.copyOf(map[i], map[i].length);
			}
			for(int i=0;i<R;i++) {
				for(int j=0;j<C;j++) {
					if(map[i][j]!=0&&map[i][j]!=-1)
						spreadDust(i,j,cmap);
				}
			}
			map=cmap;
			upPushDust();
			downDust();
			T--;
		}
		int ans=countDust();
		System.out.println(ans);
	}
	static void print() {
		for(int i=0;i<R;i++) {
			for(int j=0;j<C;j++) {
				System.out.print(map[i][j]+" ");
			}
			System.out.println();
		}
	}
	static boolean isRange(int x, int y) {
		return x>=0&&x<R&&y>=0&&y<C;
	}
	static void spreadDust(int x,int y,int cmap[][]) {
		int cnt=0;
		for(int i=0;i<4;i++) {
			int nx=x+dx[i];
			int ny=y+dy[i];
			if(isRange(nx, ny)&&map[nx][ny]!=-1) {
				cmap[nx][ny]+=(int)map[x][y]/5;
				cnt++;
			}
		}
		cmap[x][y]-=((int)map[x][y]/5)*cnt;
	}
	static void upPushDust() {
		//우
		int tmp=map[upAirx][1];
		map[upAirx][1]=0;
		int tmp2=0;
		for(int i=upAiry+2;i<C;i++) {
			tmp2=map[upAirx][i];
			map[upAirx][i]=tmp;
			tmp=tmp2;
		}
		//상
		for(int i=upAirx-1;i>=0;i--) {
			tmp2=map[i][C-1];
			map[i][C-1]=tmp;
			tmp=tmp2;
		}
		//좌
		for(int i=C-2;i>=0;i--) {
			tmp2=map[0][i];
			map[0][i]=tmp;
			tmp=tmp2;
		}
		//하
		for(int i=1;i<upAirx;i++) {
			tmp2=map[i][0];
			if(map[i][0]==-1)
				break;
			map[i][0]=tmp;
			tmp=tmp2;
		}
	}
	static void downDust() {
		//우
		int tmp=map[downAirx][1];
		map[downAirx][1]=0;
		int tmp2=0;
		for(int i=downAiry+2;i<C;i++) {
			tmp2=map[downAirx][i];
			map[downAirx][i]=tmp;
			tmp=tmp2;
		}
		//하
		for(int i=downAirx+1;i<R;i++) {
			tmp2=map[i][C-1];
			map[i][C-1]=tmp;
			tmp=tmp2;
		}
		//좌
		for(int i=C-2;i>=0;i--) {
			tmp2=map[R-1][i];
			map[R-1][i]=tmp;
			tmp=tmp2;
		}
		//상
		for(int i=R-2;i>=0;i--) {
			tmp2=map[i][0];
			if(map[i][0]==-1)
				break;
			map[i][0]=tmp;
			tmp=tmp2;
		}
	}
	static int countDust() {
		int cnt=0;
		for(int i=0;i<R;i++) {
			for(int j=0;j<C;j++) {
				if(map[i][j]!=0&&map[i][j]!=-1) {
					cnt+=map[i][j];
				}
			}
		}
		return cnt;
	}
}
```
